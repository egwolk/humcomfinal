import java.util.Scanner;
public class Calc {
    public static void main(String[]args) {
        Scanner in = new Scanner(System.in);
        
        int userIn, eq=0;
        char op;
        
        System.out.println("Welcome to calculator! (= to exit & get final value)");
        System.out.print("Enter number: ");
        userIn = in.nextInt();
        in.nextLine();
        eq += userIn;
        do{
            System.out.print("Select operator + | - | * | / | = : ");
            op = in.next().charAt(0);
            in.nextLine();
            if (op=='=') {
                break;
            }
            System.out.print("Enter number: ");
            if (op=='+') {
                eq += in.nextInt();
                in.nextLine();
                System.out.println("= "+eq );
            }
            else if (op=='-') {
                eq -= in.nextInt();
                in.nextLine();
                System.out.println("= "+eq );
            }
            else if (op=='*') {
                eq *= in.nextInt();
                in.nextLine();
                System.out.println("= "+eq );
            }
            else if (op=='/') {
                eq /= in.nextInt();
                in.nextLine();
                System.out.println("= "+eq );
            }
        }while(op!='='||op=='+'||op=='-'||op=='/'||op=='*');
        System.out.println("= "+eq);
    }
}

import java.util.Scanner;

public class exer6 {
    public static void main(String[]args) {
        
        Scanner in = new Scanner(System.in);
        
        long idNum, luck, tries=1;
        
        System.out.println("Enter ID Number:");
        idNum = in.nextLong();
        in.nextLine();	
        
        System.out.println("Guess the Lucky Number:");
        luck = in.nextLong();
        in.nextLine();
        
        while (idNum % 10 != luck) {
            System.out.println("Oops! That's not the lucky number(-_-), try again.");
            luck = in.nextInt();
            in.nextLine();
            tries++;
        }
        
        if (idNum % 10 == luck) {
            System.out.println("Congrats! You got the Lucky Number!(^_^)\n it took you "+tries+" guesses to get it!");
        }
        
    }
}

import java.util.Scanner;
public class arrays {
    public static void main(String[]args) {
        Scanner in = new Scanner(System.in);
        //input #players
        System.out.print("Enter numeber of players: ");
        int playNum = in.nextInt();
        in.nextLine();
        //make arrays
        String[] playName = new String[playNum];
        int[] playAge = new int[playNum];
        int ageSum =0;
        //store and sort array values
        for (int i = 0; i < playNum; i++) {
            System.out.print("Enter player #"+(i+1)+"'s Name: ");
            playName[i] = in.nextLine();
            System.out.print("Enter "+playName[i]+"'s Age: ");
            playAge[i] = in.nextInt();
            in.nextLine();
            
            ageSum += playAge[i];
        }
        
        System.out.println();
        //display with forloop
        System.out.println("List of players (ForLoop)");
        for (int i = 0; i < playNum; i++) {
            System.out.println("Player #"+(i+1)+"\n  Name: "+ playName[i]+" | Age: "+playAge[i]);
        }
        System.out.println();
        //display with forEachLoop
        System.out.println("List of players (ForEachLoop)");
        int temp = 0;
        for (String name : playName) {
            System.out.println("Player #"+(temp+1)+"\n Name: "+name+" | Age: "+ playAge[temp]);
            temp++;
        }
        System.out.println();
        //average age calc
        int ave = ageSum/playNum;
        System.out.println("Average player age is "+ ave);
        //display player names with below average age
        System.out.println();		
        System.out.println("Players with below average age:");
        for (int i = 0; i< playNum; i++) {
            if (ave>playAge[i]) {
                System.out.println("Player #"+(i+1)+"\n   Name: "+playName[i]+" | Age: "+playAge[i]);
                
            }
        } 
    }
}
